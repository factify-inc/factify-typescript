# AccessRequest

AccessRequest represents a request for access to a document.

## Example Usage

```typescript
import { AccessRequest } from "@factify/sdk/models/components";

let value: AccessRequest = {
  createdAt: new Date("2026-08-22T04:53:58.995Z"),
  documentId: "doc_01h2xcejqtf2nbrexx3vqjhp41",
  id: "acr_01h2xcejqtf2nbrexx3vqjhp41",
  message: "Need access for quarterly review",
  permission: "edit",
  requestStatus: "denied",
  subject: {
    id: "user_01h2xcejqtf2nbrexx3vqjhp41",
    name: "<value>",
    type: "user",
  },
  subjectEmail: "dwight@dundermifflin.com",
  updatedAt: new Date("2024-03-29T08:45:22.256Z"),
};
```
